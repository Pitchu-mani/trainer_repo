    function findSumOfDigits(number)->number            //37        
        if !(number >= 10 && number <= 99):
            // throw invalid data error 
        end if     

        firstDigit = number / 10                        // MSD // 3 
        secondDigit = number % 10                       // LSD unit digit  // 7 
        sum = firstDigit + secondDigit                  //10
        return sum
